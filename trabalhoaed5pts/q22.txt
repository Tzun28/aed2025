using System;
using System.Collections.Generic;
class Program
{
    public static void Main()
    {
        int mat = 0;
        Dictionary<int, string> Aluno = new Dictionary<int, string>();
        Console.WriteLine("Adicione uma matrícula, para sair, digite uma matrícula negativa");
        mat = int.Parse(Console.ReadLine());
        while (mat >= 0)
        {
            Console.WriteLine("Digite o nome do aluno");
            string nome = Console.ReadLine();
            Aluno[mat] = nome;
            Console.WriteLine("Adicione uma matrícula, para sair, digite uma matrícula negativa");
            mat = int.Parse(Console.ReadLine());
        }
        
        Console.WriteLine("Pesquisando:");
        Console.WriteLine("Digite a matrícula para achar o nome do aluno, negativo para sair");
        mat = int.Parse(Console.ReadLine());
        while (mat >= 0)
        {
            Console.WriteLine(Aluno[mat]);
            Console.WriteLine("Pesquisando:");
            Console.WriteLine("Digite a matrícula para achar o nome do aluno, negativo para sair");
            mat = int.Parse(Console.ReadLine());
        }
    }
}
