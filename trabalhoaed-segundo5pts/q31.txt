    class CCelulaDicionario<TK, TV> 
    { 
        public TK key; 
        public TV value; 
        public CCelulaDicionario<TK, TV> prox; 
        public CCelulaDicionario() 
        {  key = default(TK);
            value = default(TV);
            prox = null;
        } 
        public CCelulaDicionario(TK chave, TV valor)
        {
            key = chave;
            value = valor;
            prox = null;
        } 
        public CCelulaDicionario(TK chave, TV valor, CCelulaDicionario<TK, TV> proxima) 
        {    
            key = chave;
            value = valor;
            prox = proxima;
        } 
    } 
    class CDicionario<TK, TV> 
    { 
        private CCelulaDicionario<TK, TV> primeira, ultima;
        public CDicionario()
        {
            primeira = new CCelulaDicionario<TK, TV>();
            ultima = primeira;
        } 

        public bool vazio()
        {
            if (primeira == ultima){return true;}
            else {return false;}
        }
        public void adiciona(TK chave, TV valor)
        {
            CCelulaDicionario<TK,TV> aux = ultima;
            bool confereexiste = false;
            while (aux.prox != null)
            {
                aux = aux.prox;
                if (aux.key.Equals(chave))
                {
                    confereexiste = true;
                }
            }
            if (confereexiste == true)
            {
            }
            else{
                primeira.prox = new CCelulaDicionario<TK, TV>(chave, valor);
                primeira = primeira.prox;
            }
        } 

        public TV recebeValor(TK chave) 
        {
            CCelulaDicionario<TK,TV> aux = ultima; 
            while (aux.prox != null)
            {
                if (aux.prox.key.Equals(chave))
                {
                    return aux.prox.value;
                }
                aux = aux.prox;
            }
            
            Console.WriteLine("Trinca não encontrada.");
            return default(TV);
        } 
    }
